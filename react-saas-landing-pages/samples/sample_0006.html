// Deploy-aware API monitor with assertions
import { monitor } from "@pulsehook/sdk"

await monitor({
  name: "billing-api-prod",
  schedule: "*/1 * * * *",
  regions: ["iad", "fra", "sin"],
  request: {
    url: "https://api.acme.dev/v1/invoices",
    method: "GET",
    headers: { Authorization: `Bearer ${process.env.MONITOR_TOKEN}` }
  },
  expect: [
    { status: 200 },
    { jsonpath: "$.data[0].id", exists: true },
    { latencyMs: { lt: 450 } }
  ],
  alerts: ["slack:#ops", "pagerduty:primary"]
})
